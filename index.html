/* CSS Animation Example */
.animated-element {
    width: 100px;
    height: 100px;
    background: var(--primary);
    border-radius: 8px;
}

/* Bounce Animation */
@keyframes bounce {
    0%, 100% { transform: translateY(0); }
    50% { transform: translateY(-30px); }
}

.animate-bounce {
    animation: bounce 1s infinite;
}
// JavaScript Animation Example
const canvas = document.getElementById('jsAnimationCanvas');
const ctx = canvas.getContext('2d');
let x = 50;
let y = 150;
let dx = 2;
let dy = 2;
let radius = 30;
let animationId;

function draw() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    ctx.beginPath();
    ctx.arc(x, y, radius, 0, Math.PI * 2);
    ctx.fillStyle = '#4a6bff';
    ctx.fill();
    
    if (x + radius > canvas.width || x - radius < 0) dx = -dx;
    if (y + radius > canvas.height || y - radius < 0) dy = -dy;
    
    x += dx;
    y += dy;
    
    animationId = requestAnimationFrame(draw);
}
// Physics Animation Example
const ball = document.querySelector('.physics-ball');
const container = document.querySelector('.physics-container');
let yPos = 0;
let velocity = 0;
const gravity = 0.2;
const damping = 0.8;
let isAnimating = false;

function applyPhysics() {
    velocity += gravity;
    yPos += velocity;
    
    // Check for collision with bottom
    if (yPos >= container.clientHeight - ball.clientHeight) {
        yPos = container.clientHeight - ball.clientHeight;
        velocity = -velocity * damping;
        
        // Stop animation when velocity is very small
        if (Math.abs(velocity) < 0.5) {
            cancelAnimationFrame(animationId);
            isAnimating = false;
            return;
        }
    }
    
    ball.style.transform = `translateY(${yPos}px)`;
    animationId = requestAnimationFrame(applyPhysics);
}
